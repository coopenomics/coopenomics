94 void refresh_contributor_segment(eosio::name coopname, const checksum256 &project_hash, eosio::name username) {
95 Segments::segments_index segments(_capital, coopname.value);
96 auto segment_opt = Segments::get_segment(coopname, project_hash, username);
97
98 if (!segment_opt.has_value()) {
99 return; // Сегмент не найден
100 }
101
102 auto segment_it = segments.find(segment_opt->id);
103 auto project = Capital::Projects::get_project_or_fail(coopname, project_hash);
104
105 segments.modify(segment_it, _capital, [&](auto &s) {
106 // Обновляем награды вкладчика
107 if (s.capital_contributor_shares.amount > 0) {
108 int64_t pending_contributor_reward = s.capital_contributor_shares.amount *
109 (project.crps.contributor_cumulative_reward_per_share - s.last_contributor_reward_per_share);
110
111 if (pending_contributor_reward > 0) {
112 s.contributor_bonus += eosio::asset(pending_contributor_reward, _root_govern_symbol);
113 s.last_contributor_reward_per_share = project.crps.contributor_cumulative_reward_per_share;
114 }
115 }
116
117 // Обновление capital_contributor_shares на основе текущего баланса
118 eosio::asset capital_balance = Capital::Core::get_capital_user_share_balance(coopname, username);
119
120 if (s.capital_contributor_shares != capital_balance) {
121 // Корректируем общие доли в проекте
122 Capital::project_index projects(_capital, coopname.value);
123 auto project_it = projects.find(project.id);
124 projects.modify(project_it, _capital, [&](auto &p) {
125 p.crps.total_capital_contributors_shares = p.crps.total_capital_contributors_shares - s.capital_contributor_shares + capital_balance;
126 });
127
128 // Если становится новым вкладчиком, инициализируем CRPS и устанавливаем флаг
129 if (!s.is_contributor && capital_balance.amount > 0) {
130 s.is_contributor = true;
131 s.last_contributor_reward_per_share = project.crps.contributor_cumulative_reward_per_share;
132 // Увеличиваем счетчик зарегистрированных вкладчиков
133 Capital::Projects::increment_total_contributors(coopname, project_hash);
134 }
135
136 s.capital_contributor_shares = capital_balance;
137 }
138 });
