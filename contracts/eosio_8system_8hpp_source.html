42 template<typename E, typename F>
43 static inline auto has_field( F flags, E field )
44 -> std::enable_if_t< std::is_integral_v<F> && std::is_unsigned_v<F> &&
45 std::is_enum_v<E> && std::is_same_v< F, std::underlying_type_t<E> >, bool>
46 {
47 return ( (flags & static_cast<F>(field)) != 0 );
48 }
49
50 template<typename E, typename F>
51 static inline auto set_field( F flags, E field, bool value = true )
52 -> std::enable_if_t< std::is_integral_v<F> && std::is_unsigned_v<F> &&
53 std::is_enum_v<E> && std::is_same_v< F, std::underlying_type_t<E> >, F >
54 {
55 if( value )
56 return ( flags | static_cast<F>(field) );
57 else
58 return ( flags & ~static_cast<F>(field) );
59 }

184 eosio::block_signing_authority get_producer_authority()const {
185 if( producer_authority.has_value() ) {
186 bool zero_threshold = std::visit( [](auto&& auth ) -> bool {
187 return (auth.threshold == 0);
188 }, *producer_authority );
189 // zero_threshold could be true despite the validation done in regproducer2 because the v1.9.0 eosio.system
190 // contract has a bug which may have modified the producer table such that the producer_authority field
191 // contains a default constructed eosio::block_signing_authority (which has a 0 threshold and so is invalid).
192 if( !zero_threshold ) return *producer_authority;
193 }
194 return convert_to_block_signing_authority( producer_key );
195 }

197 // The unregprod and claimrewards actions modify unrelated fields of the producers table and under the default
198 // serialization behavior they would increase the size of the serialized table if the producer_authority field
199 // was not already present. This is acceptable (though not necessarily desired) because those two actions require
200 // the authority of the producer who pays for the table rows.
201 // However, the rmvproducer action and the onblock transaction would also modify the producer table in a similar
202 // way and increasing its serialized size is not acceptable in that context.
203 // So, a custom serialization is defined to handle the binary_extension producer_authority
204 // field in the desired way. (Note: v1.9.0 did not have this custom serialization behavior.)
205
206 template<typename DataStream>
207 friend DataStream& operator << ( DataStream& ds, const producer_info& t ) {
208 ds << t.owner
209 << t.total_votes
210 << t.producer_key
211 << t.is_active
212 << t.url
213 << t.unpaid_blocks
214 << t.last_claim_time
215 << t.location;
216
217 if( !t.producer_authority.has_value() ) return ds;
218
219 return ds << t.producer_authority;
220 }
221
222 template<typename DataStream>
223 friend DataStream& operator >> ( DataStream& ds, producer_info& t ) {
224 return ds >> t.owner
225 >> t.total_votes
226 >> t.producer_key
227 >> t.is_active
228 >> t.url
229 >> t.unpaid_blocks
230 >> t.last_claim_time
231 >> t.location
232 >> t.producer_authority;
233 }
234 };
