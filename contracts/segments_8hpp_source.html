273inline bool is_segment_updated(eosio::name coopname, const checksum256 &project_hash, eosio::name username) {
274 auto segment_opt = get_segment(coopname, project_hash, username);
275 if (!segment_opt.has_value()) {
276 return false; // Сегмент не найден
277 }
278
279 auto project = Capital::Projects::get_project_or_fail(coopname, project_hash);
280
281 // Проверяем актуальность CRPS для каждой роли
282 bool author_updated = (!segment_opt->is_author) ||
283 (segment_opt->last_author_base_reward_per_share == project.crps.author_base_cumulative_reward_per_share &&
284 segment_opt->last_author_bonus_reward_per_share == project.crps.author_bonus_cumulative_reward_per_share);
285
286 // Координаторы используют пропорциональное распределение на основе coordinators_investment_pool
287 bool coordinator_updated = (!segment_opt->is_coordinator) ||
288 (segment_opt->last_known_coordinators_investment_pool == project.fact.coordinators_investment_pool);
289
290 bool contributor_updated = (!segment_opt->is_contributor) ||
291 (segment_opt->last_contributor_reward_per_share == project.crps.contributor_cumulative_reward_per_share);
292
293 // Проверяем актуальность инвестиционного пула для расчета provisional_amount (нужно всем ролям)
294 bool invest_pool_updated = (segment_opt->last_known_invest_pool == project.fact.invest_pool);
295
296 // Проверяем актуальность базового пула создателей для корректного расчета использования инвестиций (нужно только инвесторам)
297 bool creators_base_pool_updated = (!segment_opt->is_investor) ||
298 (segment_opt->last_known_creators_base_pool == project.fact.creators_base_pool);
299
300 return author_updated && coordinator_updated && contributor_updated && invest_pool_updated && creators_base_pool_updated;
301}

335inline eosio::asset calculate_segment_bonus_cost(const segment& seg, const Capital::project& project) {
336 eosio::asset total = asset(0, _root_govern_symbol);
337
338 // Премии зависят от статуса проекта
339 if (project.status == Capital::Projects::Status::COMPLETED) {
340 // После голосования используем результаты голосования
341 total += seg.equal_author_bonus;
342 total += seg.direct_creator_bonus;
343 total += seg.voting_bonus;
344 } else {
345 // До голосования используем полные суммы премий
346 total += seg.creator_bonus;
347 total += seg.author_bonus;
348 }
349
350 // Премии вкладчиков
351 total += seg.contributor_bonus;
352
353 return total;
354}

369inline void update_segment_total_cost(eosio::name coopname, const checksum256 &project_hash, eosio::name username) {
370 segments_index segments(_capital, coopname.value);
371 auto idx = segments.get_index<"byprojuser"_n>();
372 auto key = combine_checksum_ids(project_hash, username);
373 auto segment_itr = idx.find(key);
374
375 eosio::check(segment_itr != idx.end(), "Сегмент участника не найден");
376
377 auto project = Capital::Projects::get_project_or_fail(coopname, project_hash);
378
379 idx.modify(segment_itr, _capital, [&](auto &s) {
380 s.total_segment_base_cost = calculate_segment_base_cost(s);
381 s.total_segment_bonus_cost = calculate_segment_bonus_cost(s, project);
382 s.total_segment_cost = s.total_segment_base_cost + s.total_segment_bonus_cost;
383 });

458 s.project_contributor_shares = asset(0, _root_govern_symbol);
