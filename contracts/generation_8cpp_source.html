23 double calculate_return_base_percent(eosio::asset creators_base_pool, eosio::asset authors_base_pool, eosio::asset coordinators_base_pool, eosio::asset invest_pool) {
24 // Только трудозатраты (без расходов)
25 double work_costs = static_cast<double>(
26 creators_base_pool.amount +
27 authors_base_pool.amount +
28 coordinators_base_pool.amount
29 );
30
31 // Если нет трудозатрат, коэффициент = 0%
32 if (work_costs == 0.0) {
33 return 0.0;
34 }
35
36 double available_investments = static_cast<double>(invest_pool.amount);
37

39 // Не может быть больше 100.0% компенсации
40 double coefficient_percent = (available_investments / work_costs) * 100.0;
41 return coefficient_percent > 100.0 ? 100.0 : coefficient_percent;

98 plan_pool calculate_plan_generation_amounts(
99 eosio::name coopname,
100 const eosio::asset& plan_hour_cost,
101 const uint64_t& plan_creators_hours,
102 const eosio::asset& plan_expenses
103 ) {
104 // читаем конфиг
105 auto st = Capital::get_global_state(coopname);
106
107 plan_pool plan;
108
109 // устанавливаем стоимость часа и время
110 plan.hour_cost = plan_hour_cost;
111 plan.creators_hours = plan_creators_hours;
112
113 // инициализируем поля расходов
114 plan.target_expense_pool = plan_expenses; // планируемые расходы
115
116 // расчет плановой себестоимости создателя (double)
117 double creator_double_amount = static_cast<double>(plan_creators_hours) * static_cast<double>(plan_hour_cost.amount);
118
119 // расчет плановой себестоимости создателя
120 plan.creators_base_pool = asset(int64_t(creator_double_amount), _root_govern_symbol);
121
122 // расчет авторской себестоимости (double)
123 double author_double_amount = static_cast<double>(creator_double_amount * AUTHOR_BASE_COEFFICIENT);
124
125 // расчет авторской себестоимости
126 plan.authors_base_pool = eosio::asset(int64_t(author_double_amount), _root_govern_symbol);
127
128 // расчет планируемых премий координаторов
129
130 plan.coordinators_base_pool = (plan.creators_base_pool + plan.authors_base_pool + plan.target_expense_pool) * st.config.coordinator_bonus_percent / 100;
131
132 // расчет премий создателя
133 double creators_bonus_double = static_cast<double>(plan.creators_base_pool.amount * CREATORS_BONUS_COEFFICIENT);
134 plan.creators_bonus_pool = eosio::asset(int64_t(creators_bonus_double), _root_govern_symbol);
135
136 // расчет премий авторов
137 double authors_bonus_double = static_cast<double>(plan.authors_base_pool.amount * AUTHOR_BONUS_COEFFICIENT);
138 plan.authors_bonus_pool = eosio::asset(int64_t(authors_bonus_double), _root_govern_symbol);
139
140 // расчет общей планируемой суммы инвестиций (все затраты)
141 plan.total_received_investments = plan.creators_base_pool + plan.authors_base_pool + plan.coordinators_base_pool + plan.target_expense_pool;
142
143 // расчет планируемой суммы инвестиций (только для трудозатрат, после вычета расходов)
144 plan.invest_pool = plan.creators_base_pool + plan.authors_base_pool + plan.coordinators_base_pool;
145
146 // расчет суммы, которую координаторы привлекут в проект
147 // считаем, что координаторы будут привлекать всю общую сумму инвестиций
148 plan.coordinators_investment_pool = plan.total_received_investments;
149
150 // коэффициенты возврата
151 plan.return_base_percent = calculate_return_base_percent(plan.creators_base_pool, plan.authors_base_pool, plan.coordinators_base_pool, plan.invest_pool);
152 plan.use_invest_percent = calculate_use_invest_percent_planned(plan.creators_base_pool, plan.authors_base_pool, plan.coordinators_base_pool, plan.target_expense_pool, plan.total_received_investments);
153
154 plan.total_generation_pool = plan.creators_base_pool + plan.authors_base_pool + plan.coordinators_base_pool + plan.creators_bonus_pool + plan.authors_bonus_pool;
155
156 // расчет премий вкладчиков
157 plan.contributors_bonus_pool = calculate_contributors_bonus_pool(plan.total_generation_pool);
158
159 // Общая планируемая сумма генерации с вкладчиками
160 plan.total = plan.total_generation_pool + plan.contributors_bonus_pool + plan.target_expense_pool;
161
162 return plan;
163 }

168 generation_amounts calculate_fact_generation_amounts(eosio::asset rate_per_hour, uint64_t creator_hours) {
169 generation_amounts fact;
170
171 // расчет фактических показателей генерации
172 fact.creators_hours = creator_hours;
173 fact.hour_cost = rate_per_hour;
174
175 // расчет себестоимости создателя
176 double creator_base_double = static_cast<double>(creator_hours) * static_cast<double>(rate_per_hour.amount);
177 fact.creators_base_pool = eosio::asset(int64_t(creator_base_double), _root_govern_symbol);
178
179 // расчет премий создателя
180 double creators_bonus_double = static_cast<double>(creator_base_double * CREATORS_BONUS_COEFFICIENT);
181 fact.creators_bonus_pool = eosio::asset(int64_t(creators_bonus_double), _root_govern_symbol);
182
183 // расчет авторской себестоимости
184 double author_base_double = static_cast<double>(creator_base_double * AUTHOR_BASE_COEFFICIENT);
185 fact.authors_base_pool = eosio::asset(int64_t(author_base_double), _root_govern_symbol); // 0.618 от creator_base
186
187 // расчет премий авторов
188 fact.authors_bonus_pool = eosio::asset(int64_t(author_base_double * AUTHOR_BONUS_COEFFICIENT), _root_govern_symbol);
189
190 // расчет общей суммы генерации коммита (без координаторов, они учитываются отдельно)
191 fact.total_generation_pool = fact.creators_base_pool + fact.authors_base_pool + fact.creators_bonus_pool + fact.authors_bonus_pool;
192
193 // Расчет премий вкладчиков от общей суммы генерации коммита
194 fact.contributors_bonus_pool = eosio::asset(int64_t(static_cast<double>(
195 fact.total_generation_pool.amount
196 ) * CONTRIBUTORS_BONUS_COEFFICIENT), _root_govern_symbol);
197
198 // Расчет общей суммы вкладов всех пайщиков (генерация + премии вкладчиков)
199 fact.total_contribution = fact.total_generation_pool + fact.contributors_bonus_pool;
200
201 return fact;
202 }

262 void add_coordinator_funds(eosio::name coopname, const checksum256 &project_hash, const eosio::asset &amount) {
263 auto exist_project = Capital::Projects::get_project_or_fail(coopname, project_hash);

290 });

292 // Примечание: Координаторские награды будут распределены пропорционально при обновлении сегментов
