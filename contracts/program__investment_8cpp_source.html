38 eosio::asset expense_gap = p.plan.target_expense_pool - p.fact.accumulated_expense_pool;

47
48 // Но не больше, чем нужно для достижения цели

51
52 // Остальные средства идут в инвестиционные пулы
53 eosio::asset to_invest_pool = amount - to_expense_pool;

60 p.fact.accumulated_expense_pool += to_expense_pool;
