46 void refresh_provisional_amount(eosio::name coopname, const checksum256 &project_hash, eosio::name username) {
47 auto segment_opt = Segments::get_segment(coopname, project_hash, username);
48 if (!segment_opt.has_value()) {
49 return; // Сегмент не найден
50 }
51
52 auto project = Capital::Projects::get_project_or_fail(coopname, project_hash);
53
54 // Базовые поля текущего пользователя (себестоимость труда)
55 int64_t user_base_amount = segment_opt->creator_base.amount +
56 segment_opt->author_base.amount +
57 segment_opt->coordinator_base.amount; //имущественный взнос здесь НЕ учитывается
58
59 // Если нет себестоимости труда у пользователя, provisional_amount = 0
60 if (user_base_amount == 0) {
61 Segments::segments_index segments(_capital, coopname.value);
62 auto segment_it = segments.find(segment_opt->id);
63 segments.modify(segment_it, _capital, [&](auto &s) {
64 s.provisional_amount = eosio::asset(0, _root_govern_symbol);
65 s.last_known_invest_pool = project.fact.invest_pool; // Все равно синхронизируем инвестиции
66 s.last_known_creators_base_pool = project.fact.creators_base_pool; // Синхронизируем с актуальной суммой базового пула создателей
67 });
68 return;
69 }
70
71 // Рассчитываем provisional_amount с учетом return_base_percent (в процентах)
72 double return_coefficient_percent = project.fact.return_base_percent;
73
74 // provisional_amount = себестоимость_труда * (return_base_percent / 100)
75 int64_t provisional_amount = static_cast<int64_t>(
76 static_cast<double>(user_base_amount) * (return_coefficient_percent / 100.0)
77 );
78
79 eosio::check(provisional_amount <= project.fact.invest_pool.amount, "Cумма доступной ссуды не может превышать сумму инвестиций проекта");
80
81 // Обновляем provisional_amount и синхронизируем известные пулы в сегменте
82 Segments::segments_index segments(_capital, coopname.value);
83 auto segment_it = segments.find(segment_opt->id);
84 segments.modify(segment_it, _capital, [&](auto &s) {
85 s.provisional_amount = eosio::asset(provisional_amount, _root_govern_symbol);
86 s.last_known_invest_pool = project.fact.invest_pool; // Синхронизируем с актуальной суммой инвестиций
87 s.last_known_creators_base_pool = project.fact.creators_base_pool; // Синхронизируем с актуальной суммой базового пула создателей
88 });
89 }

96 void upsert_investor_segment(eosio::name coopname, const checksum256 &project_hash,
97 eosio::name username, const eosio::asset &investor_amount) {
98 Segments::segments_index segments(_capital, coopname.value);
99 auto exist_segment = Segments::get_segment(coopname, project_hash, username);
100
101 if (!exist_segment.has_value()) {
102 segments.emplace(_capital, [&](auto &g){
103 g.id = segments.available_primary_key();
104 g.coopname = coopname;
105 g.project_hash = project_hash;
106 g.username = username;
107 g.investor_amount = investor_amount;
108 g.is_investor = true;
109 });
110
111 Capital::Projects::increment_total_investors(coopname, project_hash);
112
113 } else {
114 auto segment = segments.find(exist_segment->id);
115 segments.modify(segment, _capital, [&](auto &g) {
116 if (!g.is_investor) {
117 g.is_investor = true;
118 Capital::Projects::increment_total_investors(coopname, project_hash);
119 }
120 g.investor_amount += investor_amount;
121 });
122 }
123
124 // Обновляем общую стоимость сегмента
125 Capital::Segments::update_segment_total_cost(coopname, project_hash, username);
126 }

131 void refresh_project_wallet_membership_rewards(eosio::name coopname, const checksum256 &project_hash, eosio::name username) {
132 auto project = Capital::Projects::get_project_or_fail(coopname, project_hash);
133 auto wallet_opt = Capital::get_project_wallet(coopname, project_hash, username);
134
135 if (!wallet_opt.has_value()) {
136 return; // Кошелек проекта не найден
137 }
138
139 Capital::project_wallets_index project_wallets(_capital, coopname.value);
140 auto wallet_it = project_wallets.find(wallet_opt->id);
141
142 // Если у участника нет долей в кошельке проекта, то нет и права на членские взносы
143 if (wallet_opt->shares.amount == 0 || project.membership.total_shares.amount == 0) {
144 project_wallets.modify(wallet_it, _capital, [&](auto &w) {
145 w.last_membership_reward_per_share = project.membership.cumulative_reward_per_share;
146 });
147 return;
148 }
149
150 project_wallets.modify(wallet_it, _capital, [&](auto &w) {
151 // Рассчитываем накопленные награды от членских взносов на основе долей в кошельке проекта
152 int64_t pending_membership_reward = w.shares.amount *
153 (project.membership.cumulative_reward_per_share - w.last_membership_reward_per_share);
154
155 if (pending_membership_reward > 0) {
156 w.membership_available += eosio::asset(pending_membership_reward, _root_govern_symbol);
157 }
158
159 w.last_membership_reward_per_share = project.membership.cumulative_reward_per_share;
160 });
161 }
