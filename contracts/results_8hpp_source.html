113inline void create_result_for_participant(eosio::name coopname, const checksum256 &project_hash,
114 eosio::name username, const checksum256 &result_hash,
115 eosio::asset segment_cost, eosio::asset debt_amount, const document2 &statement) {
116 result_index results(_capital, coopname.value);
117
118 // Проверяем, что результат ещё не создан
119 auto existing = get_result_by_project_and_username(coopname, project_hash, username);
120 eosio::check(!existing.has_value(), "Результат уже создан для данного участника");
121
122 results.emplace(_capital, [&](auto &r) {
123 r.id = results.available_primary_key();
124 r.project_hash = project_hash;
125 r.result_hash = result_hash;
126 r.coopname = coopname;
127 r.username = username;
128 r.status = Capital::Results::Status::CREATED;
129 r.total_amount = segment_cost;
130 r.debt_amount = debt_amount;
131 r.statement = statement;
132 r.created_at = current_time_point();
133 });
134}

140inline void delete_result_and_reset_segment(eosio::name coopname, const checksum256 &project_hash,
