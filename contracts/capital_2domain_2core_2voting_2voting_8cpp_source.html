33 voting_amounts calculate_voting_amounts(const eosio::asset& authors_bonus_pool,
34 const eosio::asset& creators_bonus_pool,
35 uint64_t total_authors,
36 uint32_t total_voters,
37 double authors_voting_percent,
38 double creators_voting_percent) {
39 voting_amounts result;
40
41 // Коэффициенты
42 double authors_equal_percent = (100.0 - authors_voting_percent) / 100.0;
43 double creators_direct_percent = (100.0 - creators_voting_percent) / 100.0;
44 double authors_voting_coeff = authors_voting_percent / 100.0;
45 double creators_voting_coeff = creators_voting_percent / 100.0;
46
47 // Авторские премии
48 result.authors_equal_spread = eosio::asset(
49 int64_t(static_cast<double>(authors_bonus_pool.amount) * authors_equal_percent),
50 authors_bonus_pool.symbol
51 );
52 result.authors_bonuses_on_voting = eosio::asset(
53 int64_t(static_cast<double>(authors_bonus_pool.amount) * authors_voting_coeff),
54 authors_bonus_pool.symbol
55 );
56
57 if (total_authors > 0) {
58 result.authors_equal_per_author = eosio::asset(
59 result.authors_equal_spread.amount / total_authors,
60 result.authors_equal_spread.symbol
61 );
62 } else {
63 result.authors_equal_per_author = eosio::asset(0, authors_bonus_pool.symbol);
64 }
65
66 // Создательские премии
67 result.creators_direct_spread = eosio::asset(
68 int64_t(static_cast<double>(creators_bonus_pool.amount) * creators_direct_percent),
69 creators_bonus_pool.symbol
70 );
71 result.creators_bonuses_on_voting = eosio::asset(
72 int64_t(static_cast<double>(creators_bonus_pool.amount) * creators_voting_coeff),
73 creators_bonus_pool.symbol
74 );
75
76 // Общий пул для голосования
77 result.total_voting_pool = result.authors_bonuses_on_voting + result.creators_bonuses_on_voting;
78
79 // Активная голосующая сумма
80 result.active_voting_amount = eosio::asset(
81 int64_t(static_cast<double>(result.total_voting_pool.amount) * (total_voters - 1) / total_voters),
82 result.total_voting_pool.symbol
83 );
84
85 // Равная сумма на каждого
86 result.equal_voting_amount = eosio::asset(
87 int64_t(static_cast<double>(result.total_voting_pool.amount) / total_voters),
88 result.total_voting_pool.symbol
89 );
90
91 // ---- Корректировка хвостика ----
92 int64_t distributed = result.equal_voting_amount.amount * total_voters;
93 int64_t diff = result.total_voting_pool.amount - distributed;
94 if (diff != 0) {
95 result.equal_voting_amount.amount += diff;
96 }
97
98 return result;
99 }

102 eosio::asset calculate_voting_final_amount(name coopname, checksum256 project_hash, name participant) {
103 auto project = Capital::Projects::get_project_or_fail(coopname, project_hash);
104
105 // Получаем сегмент участника для определения его ролей
106 auto segment = Capital::Segments::get_segment_or_fail(coopname, project_hash, participant, "Сегмент участника не найден");
107
108 // Получаем все голоса за данного участника от других участников
109 auto votes = Capital::Votes::get_votes_for_recipient(coopname, project_hash, participant);
110
111 // Рассчитываем общую сумму голосов от других участников
112 eosio::asset total_votes_received = asset(0, project.voting.amounts.total_voting_pool.symbol);
113 for (const auto& vote : votes) {
114 total_votes_received += vote.amount;
115 }
116
117 // Формула: (сумма всех полученных голосов + средняя сумма на каждого) / общее количество голосующих
118 eosio::asset total_sum = (total_votes_received + project.voting.amounts.equal_voting_amount) / project.voting.total_voters;
119
120 // Итоговая сумма от голосования по Водянову
121 return total_sum;
122 }

157 void update_voting_status(eosio::name coopname, const checksum256 &project_hash, eosio::name username) {
158 Capital::Segments::segments_index segments(_capital, coopname.value);
159 auto idx = segments.get_index<"byprojuser"_n>();
160 auto key = combine_checksum_ids(project_hash, username);
161 auto segment_itr = idx.find(key);
162
163 if (segment_itr == idx.end()) {
164 return; // Сегмент не найден
165 }
166
167 bool had_vote_before = segment_itr->has_vote;
168 bool should_have_vote = segment_itr->is_author || segment_itr->is_creator;
169
170 // Обновляем статус голосования только если участник получил право голоса
171 if (!had_vote_before && should_have_vote) {
172 idx.modify(segment_itr, _capital, [&](auto &s) {
173 s.has_vote = true;
174 });
175
176 // Участник получил право голоса - увеличиваем счетчик
177 Capital::Projects::increment_total_voters(coopname, project_hash);
178 }
179 // Участник не может потерять право голоса - убираем эту логику
180 }
