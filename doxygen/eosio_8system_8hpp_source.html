69 template<typename E, typename F>
70 static inline auto has_field( F flags, E field )
71 -> std::enable_if_t< std::is_integral_v<F> && std::is_unsigned_v<F> &&
72 std::is_enum_v<E> && std::is_same_v< F, std::underlying_type_t<E> >, bool>
73 {
74 return ( (flags & static_cast<F>(field)) != 0 );
75 }
76
77 template<typename E, typename F>
78 static inline auto set_field( F flags, E field, bool value = true )
79 -> std::enable_if_t< std::is_integral_v<F> && std::is_unsigned_v<F> &&
80 std::is_enum_v<E> && std::is_same_v< F, std::underlying_type_t<E> >, F >
81 {
82 if( value )
83 return ( flags | static_cast<F>(field) );
84 else
85 return ( flags & ~static_cast<F>(field) );
86 }

234 eosio::block_signing_authority get_producer_authority()const {
235 if( producer_authority.has_value() ) {
236 bool zero_threshold = std::visit( [](auto&& auth ) -> bool {
237 return (auth.threshold == 0);
238 }, *producer_authority );
239 // zero_threshold could be true despite the validation done in regproducer2 because the v1.9.0 eosio.system
240 // contract has a bug which may have modified the producer table such that the producer_authority field
241 // contains a default constructed eosio::block_signing_authority (which has a 0 threshold and so is invalid).
242 if( !zero_threshold ) return *producer_authority;
243 }
244 return convert_to_block_signing_authority( producer_key );
245 }

247 // The unregprod and claimrewards actions modify unrelated fields of the producers table and under the default
248 // serialization behavior they would increase the size of the serialized table if the producer_authority field
249 // was not already present. This is acceptable (though not necessarily desired) because those two actions require
250 // the authority of the producer who pays for the table rows.
251 // However, the rmvproducer action and the onblock transaction would also modify the producer table in a similar
252 // way and increasing its serialized size is not acceptable in that context.
253 // So, a custom serialization is defined to handle the binary_extension producer_authority
254 // field in the desired way. (Note: v1.9.0 did not have this custom serialization behavior.)
255
256 template<typename DataStream>
257 friend DataStream& operator << ( DataStream& ds, const producer_info& t ) {
258 ds << t.owner
259 << t.total_votes
260 << t.producer_key
261 << t.is_active
262 << t.url
263 << t.unpaid_blocks
264 << t.last_result_time
265 << t.location;
266
267 if( !t.producer_authority.has_value() ) return ds;
268
269 return ds << t.producer_authority;
270 }
271
272 template<typename DataStream>
273 friend DataStream& operator >> ( DataStream& ds, producer_info& t ) {
274 return ds >> t.owner
275 >> t.total_votes
276 >> t.producer_key
277 >> t.is_active
278 >> t.url
279 >> t.unpaid_blocks
280 >> t.last_result_time
281 >> t.location
282 >> t.producer_authority;
283 }
